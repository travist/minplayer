1 /** The minplayer namespace. */
  2 minplayer = minplayer || {};
  3 
  4 /**
  5  * @constructor
  6  * @extends minplayer.plugin
  7  * @class Base class used to provide the display and options for any component
  8  * deriving from this class.  Components who derive are expected to provide
  9  * the elements that they define by implementing the getElements method.
 10  *
 11  * @param {string} name The name of this plugin.
 12  * @param {object} context The jQuery context this component resides.
 13  * @param {object} options The options for this component.
 14  */
 15 minplayer.display = function(name, context, options) {
 16 
 17   // See if we allow resize on this display.
 18   this.allowResize = false;
 19 
 20   if (context) {
 21 
 22     // Set the display and options.
 23     this.display = jQuery(context);
 24     this.options = options;
 25 
 26     // Extend all display elements.
 27     this.options.elements = this.options.elements || {};
 28     jQuery.extend(this.options.elements, this.getElements());
 29     this.elements = this.options.elements;
 30   }
 31 
 32   // Derive from plugin
 33   minplayer.plugin.call(this, name, context, options);
 34 };
 35 
 36 /** Derive from minplayer.plugin. */
 37 minplayer.display.prototype = new minplayer.plugin();
 38 
 39 /** Reset the constructor. */
 40 minplayer.display.prototype.constructor = minplayer.display;
 41 
 42 /**
 43  * @see minplayer.plugin.construct
 44  */
 45 minplayer.display.prototype.construct = function() {
 46 
 47   // Call the plugin constructor.
 48   minplayer.plugin.prototype.construct.call(this);
 49 
 50   // Only do this if they allow resize for this display.
 51   if (this.allowResize) {
 52 
 53     // Set the resize timeout and this pointer.
 54     var resizeTimeout = 0;
 55     var _this = this;
 56 
 57     // Add a handler to trigger a resize event.
 58     jQuery(window).resize(function() {
 59       clearTimeout(resizeTimeout);
 60       resizeTimeout = setTimeout(function() {
 61         _this.onResize();
 62       }, 200);
 63     });
 64   }
 65 };

 77  * @param {string} type The event type.
 78  * @param {object} data The event data object.
 79  * @return {object} The jQuery prototype.

 81 minplayer.display.prototype.trigger = function(type, data) {

 83 };

 85 /**
 86  * Bind to a media event.
 87  *
 88  * @param {string} types The event type.
 89  * @param {object} data The data to bind with the event.
 90  * @param {function} fn The callback function.
 91  * @return {object} The jQuery prototype.
 92  **/
 93 minplayer.display.prototype.bind = function(types, data, fn) {
 94 
 95   // We will always unbind first for media events.
 96   return this.display.unbind(types, fn).bind(types, data, fn);
 97 };

 99 /**
100  * Returns a scaled rectangle provided a ratio and the container rect.
101  *
102  * @param {number} ratio The width/height ratio of what is being scaled.
103  * @param {object} rect The bounding rectangle for scaling.
104  * @return {object} The Rectangle object of the scaled rectangle.
105  */
106 minplayer.display.prototype.getScaledRect = function(ratio, rect) {
107   var scaledRect = {};
108   scaledRect.x = rect.x ? rect.x : 0;
109   scaledRect.y = rect.y ? rect.y : 0;
110   scaledRect.width = rect.width ? rect.width : 0;
111   scaledRect.height = rect.height ? rect.height : 0;
112   if (ratio) {
113     if ((rect.width / rect.height) > ratio) {
114       scaledRect.height = rect.height;
115       scaledRect.width = Math.floor(rect.height * ratio);
116     }
117     else {
118       scaledRect.height = Math.floor(rect.width / ratio);
119       scaledRect.width = rect.width;
120     }
121     scaledRect.x = Math.floor((rect.width - scaledRect.width) / 2);
122     scaledRect.y = Math.floor((rect.height - scaledRect.height) / 2);
123   }
124   return scaledRect;
125 };
